import java.util.Arrays;
import java.util.Comparator;

public class FindBigNumber {

    public String solution(int[] numbers) {

        String[] str = new String[numbers.length];
        String answer = "";

        // 문자열로 치환
        for (int i = 0; i < numbers.length; i++) {
            str[i] = String.valueOf(numbers[i]);
        }

        // 음수나 0을 리턴하면 자리 그대로 유지, 양수일 때는 두 객체의 자리가 변경
        Arrays.sort(str, new Comparator<String>() {
            @Override
            public int compare(String o1, String o2) {
                return (o2 + o1).compareTo(o1 + o2);
            }
        });


        for (String val : str) {
            System.out.println(val);
        }

        if (str[0].startsWith("0")) {
            answer += "0";
        } else {
            for (int i = 0; i < str.length; i++) {
                answer += str[i];
            }

        }

        return answer;
    }


    public static void main(String[] args) {

        FindBigNumber findBigNumber = new FindBigNumber();

        int[] numbers = {6, 10, 2};
        int[] numbers2 = {3, 30, 34, 5, 9};

        String result = findBigNumber.solution(numbers);
        String result2 = findBigNumber.solution(numbers2);


        System.out.println("result: " + result);
        System.out.println("result: " + result2);
    }


}
